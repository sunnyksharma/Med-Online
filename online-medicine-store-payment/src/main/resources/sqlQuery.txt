DROP DATABASE if exists EPharmacy_PaymentMS_DB;
CREATE DATABASE EPharmacy_PaymentMS_DB;
USE EPharmacy_PaymentMS_DB;

CREATE TABLE CARD(
	CARD_ID VARCHAR(16) PRIMARY KEY,
	NAME_ON_CARD VARCHAR(50),
	CVV VARCHAR(70),
	CARD_TYPE ENUM('DEBIT_CARD','CREDIT_CARD'),
	EXPIRY_DATE DATE,
	CUSTOMER_ID INT
);

CREATE TABLE PAYMENT(
	PAYMENT_ID INT PRIMARY KEY AUTO_INCREMENT,
	PAYMENT_TIME DATETIME,
	AMOUNT DECIMAL(10,2),
	CARD_ID VARCHAR(16) REFERENCES CARD(CARD_ID),
	CUSTOMER_ID INT
);


commit;
